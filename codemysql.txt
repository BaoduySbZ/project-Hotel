CREATE TABLE `comment` (
  `id` INT NOT NULL AUTO_INCREMENT,
  `userId` INT NOT NULL,
  `roomId` INT,
  `hotelId` INT,
  `rating` DECIMAL(2, 1) DEFAULT 0.0,
  `comment` TEXT,
  `is_removed` TINYINT DEFAULT 0,
  PRIMARY KEY (`id`),
  FOREIGN KEY (`userId`) REFERENCES `account`(`id`) ON DELETE CASCADE ON UPDATE CASCADE,
  FOREIGN KEY (`roomId`) REFERENCES `rooms`(`id`) ON DELETE SET NULL ON UPDATE CASCADE,
  FOREIGN KEY (`hotelId`) REFERENCES `hotel`(`id`) ON DELETE SET NULL ON UPDATE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_0900_ai_ci;
